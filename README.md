    ];

    const bgText = document.getElementById('bgText');
    let currentIndex = -1;

    function updateBackground(index) {
      if (index !== currentIndex) {
        currentIndex = index;
        bgText.classList.add('hide');

        setTimeout(() => {
          bgText.textContent = messages[index].text;
          bgText.className = `bg-glow-text ${messages[index].className}`;
          document.body.style.background = messages[index].bgGradient;
          bgText.classList.remove('hide');
        }, 400);
      }
    }

    updateBackground(0);

    window.addEventListener('scroll', () => {
      const vh = window.innerHeight;
      const scrollPos = window.scrollY;
      const index = Math.min(Math.floor(scrollPos / vh), messages.length - 1);
      updateBackground(index);
    });
